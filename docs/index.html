compiler_demo.py 

from compiler import CompilerPipeline

# Generic DSL Engine with Pratt Parser & Trace Tree
pipeline = CompilerPipeline(
    tokenizer=RuleTokenizer(),
    parser_factory=lambda t: RuleParser(t),
    executor=RuleExecutor()
)

result, trace = pipeline.run(
    source='IF age > 18 AND status == "active" THEN approved',
    context={"age": 25, "status": "active", "approved": True}
)
# >> result: True
